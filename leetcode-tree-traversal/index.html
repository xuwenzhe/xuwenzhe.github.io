# class Solution(object):
# 	def inorderTraversal(self, root):
# 		"""
# 		:type root: TreeNode
# 		:rtype: List[int]
# 		"""
		if root is None:
			return []
		dummy = TreeNode(0)
		dummy.right = root
		stack = [dummy]
			
		inorder = []
		while stack:
			node = stack.pop()
			if node.right:
				# 想处理此节点，先把此节点的右子树加入优先级栈
				node = node.right
				while node:
					stack.append(node)
					node = node.left
			if stack:
				inorder.append(stack[-1].val)
				
		return inorder
